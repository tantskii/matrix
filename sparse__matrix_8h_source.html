46 template <typename T, T Default, size_t N>
47 void Matrix<T, Default, N>::update(Indexes&& indexes, T&& value) {
48  /*
49  1. Если пришло значение по умолчанию и элемент с такими индексами существует
50  -- удаляем этот элемент
51  2. Если пришло значение по умолчанию и элемент с такими индексами НЕ существует
52  -- ничего не делаем
53  3. Если пришло НЕ значение по умолчанию и элемент с такими индексами НЕ существует
54  -- сохраняем этот элемент
55  4. Если пришло НЕ значение по умолчанию и элемент с такими индексами существует
56  -- удаляем старое значение по этому ключу и затем добавляем новое с таким же ключом
57  */
58 
59  const auto key = m_data.makeKey(indexes);
60  const bool is_default = value == Default;
61  const bool exists = m_data.contains(key);
62 
63  if (is_default && exists) {
64  // п.1 Значение по умолчанию и элемент с такими индексами существует
65  m_data.erase(key);
66  } else if (!is_default) {
67  // п.3 & п.4 Пришло НЕ значение по умолчанию и элемент
68  m_data.insert(key, value); // Data<T, N>::insert удалит старое значение,
69  // в случае, когда элемент по такому ключу уже существует
70  }
71 }
